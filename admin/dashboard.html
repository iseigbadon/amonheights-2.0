let properties = []; let filteredProperties = []; let currentPage = 1; let currentFilter = 'all'; let currentSort = 'recent'; let sortColumn = null; let sortDirection = 'asc'; let propertyToDelete = null; let editingPropertyId = null; /* ============================================ INITIALIZATION ============================================ */ document.addEventListener('DOMContentLoaded', function() { // CHECK AUTHENTICATION const isLoggedIn = sessionStorage.getItem('adminLoggedIn'); const adminUser = sessionStorage.getItem('adminUsername'); if (!isLoggedIn || !adminUser) { // Redirect to login if not authenticated window.location.href = './login.html'; return; } // User is authenticated - load dashboard loadProperties(); setupEventListeners(); }); function setupEventListeners() { // Tab navigation document.querySelectorAll('.tab-btn').forEach(btn => { btn.addEventListener('click', function() { switchTab(this.dataset.tab); }); }); // Property form submission document.getElementById('propertyForm').addEventListener('submit', function(e) { e.preventDefault(); addProperty(); }); // Property search/filter document.getElementById('propertyFilter').addEventListener('input', function() { currentPage = 1; applyFilters(); }); // Filter buttons document.querySelectorAll('.filter-btn').forEach(btn => { btn.addEventListener('click', function() { document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active')); this.classList.add('active'); currentFilter = this.dataset.filter; currentPage = 1; applyFilters(); }); }); // Sort dropdown document.getElementById('sortBy').addEventListener('change', function() { currentSort = this.value; currentPage = 1; applyFilters(); }); // Sortable table headers document.querySelectorAll('th.sortable').forEach(th => { th.addEventListener('click', function() { const column = this.dataset.sort; if (sortColumn === column) { sortDirection = sortDirection === 'asc' ? 'desc' : 'asc'; } else { sortColumn = column; sortDirection = 'asc'; } document.querySelectorAll('th.sortable').forEach(h => { h.classList.remove('asc', 'desc'); }); this.classList.add(sortDirection); currentPage = 1; applyFilters(); }); }); // Edit modal close document.getElementById('editModal').addEventListener('click', function(e) { if (e.target === this) closeEditModal(); }); // Delete modal close document.getElementById('deleteModal').addEventListener('click', function(e) { if (e.target === this) closeDeleteModal(); }); // Logout document.getElementById('logoutBtn').addEventListener('click', function() { if (confirm('Are you sure you want to logout?')) { window.location.href = './login.html'; } }); } /* ============================================ UTILITIES ============================================ */ function sanitizeInput(input) { const div = document.createElement('div'); div.textContent = input; return div.innerHTML; } function showLoading(show = true) { document.getElementById('loadingOverlay').classList.toggle('show', show); } function showMessage(elementId, message, type) { const messageEl = document.getElementById(elementId); messageEl.textContent = message; messageEl.className = `message show ${type}`; setTimeout(() => { messageEl.classList.remove('show'); }, 4000); } /* ============================================ STORAGE OPERATIONS ============================================ */ function loadProperties() { showLoading(true); setTimeout(() => { try { const stored = localStorage.getItem(STORAGE_KEY); properties = stored ? JSON.parse(stored) : []; } catch (error) { console.error('Error loading from storage:', error); properties = []; } applyFilters(); showLoading(false); }, 500); } function saveProperties() { try { localStorage.setItem(STORAGE_KEY, JSON.stringify(properties)); } catch (error) { console.error('Error saving to storage:', error); showMessage('formMessage', 'Error saving property', 'error'); } } function getNextPropertyId() { return properties.length === 0 ? 1 : Math.max(...properties.map(p => p.id)) + 1; } /* ============================================ TAB SWITCHING ============================================ */ function switchTab(tabId) { document.querySelectorAll('.tab-content').forEach(tab => { tab.classList.remove('active'); }); document.querySelectorAll('.tab-btn').forEach(btn => { btn.classList.remove('active'); }); document.getElementById(tabId).classList.add('active'); document.querySelector(`[data-tab="${tabId}"]`).classList.add('active'); } /* ============================================ FILTERING & SORTING ============================================ */ function applyFilters() { let filtered = [...properties]; // Apply status filter if (currentFilter !== 'all') { filtered = filtered.filter(p => p.status === currentFilter); } // Apply search filter const searchTerm = document.getElementById('propertyFilter').value.toLowerCase(); if (searchTerm) { filtered = filtered.filter(p => (p.name && p.name.toLowerCase().includes(searchTerm)) || (p.location && p.location.toLowerCase().includes(searchTerm)) || (p.price && p.price.toString().includes(searchTerm)) ); } // Apply sorting filtered = applySorting(filtered); filteredProperties = filtered; currentPage = 1; displayPaginatedProperties(); } function applySorting(arr) { const sorted = [...arr]; if (sortColumn) { sorted.sort((a, b) => { let aVal = a[sortColumn]; let bVal = b[sortColumn]; if (typeof aVal === 'string') { aVal = aVal.toLowerCase(); bVal = bVal.toLowerCase(); } const comparison = aVal < bVal ? -1 : aVal > bVal ? 1 : 0; return sortDirection === 'asc' ? comparison : -comparison; }); } else { // Apply dropdown sort sorted.sort((a, b) => { switch(currentSort) { case 'priceHigh': return b.price - a.price; case 'priceLow': return a.price - b.price; case 'nameAZ': return a.name.localeCompare(b.name); case 'nameZA': return b.name.localeCompare(a.name); default: // recent return b.id - a.id; } }); } return sorted; } /* ============================================ PAGINATION ============================================ */ function displayPaginatedProperties() { const startIdx = (currentPage - 1) * ITEMS_PER_PAGE; const endIdx = startIdx + ITEMS_PER_PAGE; const pageData = filteredProperties.slice(startIdx, endIdx); displayProperties(pageData); displayPagination(); } function displayPagination() { const totalPages = Math.ceil(filteredProperties.length / ITEMS_PER_PAGE); const paginationContainer = document.getElementById('paginationContainer'); const paginationInfo = document.getElementById('paginationInfo'); if (totalPages <= 1) { paginationContainer.style.display = 'none'; paginationInfo.textContent = filteredProperties.length === 0 ? 'No properties found' : `Showing ${filteredProperties.length} properties`; return; } paginationContainer.style.display = 'flex'; paginationContainer.innerHTML = ''; // Previous button const prevBtn = document.createElement('button'); window.scrollTo({top: 0, behavior: 'smooth'}); } }; paginationContainer.appendChild(prevBtn); // Page numbers for (let i = 1; i <= totalPages; i++) { const pageBtn = document.createElement('button'); pageBtn.className = 'pagination-btn'; if (i === currentPage) pageBtn.classList.add('active'); pageBtn.textContent = i; pageBtn.onclick = () => { currentPage = i; displayPaginatedProperties(); window.scrollTo({top: 0, behavior: 'smooth'}); }; paginationContainer.appendChild(pageBtn); } // Next button const nextBtn = document.createElement('button'); nextBtn.className = 'pagination-btn'; nextBtn.textContent = 'Next →'; nextBtn.disabled = currentPage === totalPages; nextBtn.onclick = () => { if (currentPage < totalPages) { currentPage++; displayPaginatedProperties(); window.scrollTo({top: 0, behavior: 'smooth'}); } }; paginationContainer.appendChild(nextBtn); const startIdx = (currentPage - 1) * ITEMS_PER_PAGE + 1; const endIdx = Math.min(currentPage * ITEMS_PER_PAGE, filteredProperties.length); paginationInfo.textContent = `Showing ${startIdx}-${endIdx} of ${filteredProperties.length} properties`; } function displayProperties(propertiesArray) { const tbody = document.getElementById('propertiesTableBody'); const emptyState = document.getElementById('emptyState'); if (propertiesArray.length === 0) { tbody.innerHTML = 'No properties found'; if (filteredProperties.length === 0) { emptyState.style.display = 'block'; } return; } emptyState.style.display = 'none'; tbody.innerHTML = propertiesArray.map(property => ` ${sanitizeInput(property.name || 'N/A')} ${sanitizeInput(property.location || 'N/A')} ${sanitizeInput(property.type || 'N/A')} ₦${parseInt(property.price || 0).toLocaleString()} ${sanitizeInput(property.status || 'N/A')}
`).join(''); } /* ============================================ CRUD OPERATIONS ============================================ */ function addProperty() { const form = document.getElementById('propertyForm'); const formData = new FormData(form); const propertyData = Object.fromEntries(formData); // Validation if (!propertyData.name || !propertyData.location || !propertyData.type || !propertyData.price || !propertyData.status) { showMessage('formMessage', '⚠️ Please fill all required fields', 'error'); return; } // Sanitize inputs Object.keys(propertyData).forEach(key => { if (typeof propertyData[key] === 'string') { propertyData[key] = sanitizeInput(propertyData[key]); } }); const newProperty = { id: getNextPropertyId(), ...propertyData, price: parseInt(propertyData.price), bedrooms: parseInt(propertyData.bedrooms) || 0, bathrooms: parseInt(propertyData.bathrooms) || 0, sqft: parseInt(propertyData.sqft) || 0, rating: parseFloat(propertyData.rating) || 0, reviews: parseInt(propertyData.reviews) || 0, amenities: propertyData.amenities ? propertyData.amenities.split(',').map(a => a.trim()) : [] }; properties.push(newProperty); saveProperties(); form.reset(); showMessage('formMessage', '✅ Property added successfully!', 'success'); setTimeout(() => { switchTab('properties-tab'); applyFilters(); }, 1000); } function openEditModal(propertyId) { const property = properties.find(p => p.id === propertyId); if (!property) return; editingPropertyId = propertyId; document.getElementById('editPropertyId').value = propertyId; document.getElementById('editPropertyName').value = property.name; document.getElementById('editPropertyLocation').value = property.location; document.getElementById('editPropertyType').value = property.type; document.getElementById('editPropertyStatus').value = property.status; document.getElementById('editPropertyPrice').value = property.price; document.getElementById('editPropertyRating').value = property.rating || ''; document.getElementById('editPropertyBedrooms').value = property.bedrooms || ''; document.getElementById('editPropertyBathrooms').value = property.bathrooms || ''; document.getElementById('editPropertySize').value = property.sqft || ''; document.getElementById('editPropertyReviews').value = property.reviews || ''; document.getElementById('editPropertyDescription').value = property.description || ''; document.getElementById('editPropertyImage').value = property.image || ''; document.getElementById('editPropertyAmenities').value = property.amenities ? property.amenities.join(', ') : ''; document.getElementById('editModal').classList.add('show'); } function closeEditModal() { document.getElementById('editModal').classList.remove('show'); editingPropertyId = null; } function savePropertyEdit() { if (editingPropertyId === null) return; const form = document.getElementById('editPropertyForm'); const formData = new FormData(form); const updated = Object.fromEntries(formData); // Validation if (!updated.name || !updated.location || !updated.type || !updated.price || !updated.status) { showMessage('propertiesMessage', '⚠️ Please fill all required fields', 'error'); return; } // Sanitize Object.keys(updated).forEach(key => { if (typeof updated[key] === 'string') { updated[key] = sanitizeInput(updated[key]); } }); const propertyIdx = properties.findIndex(p => p.id === editingPropertyId); if (propertyIdx !== -1) { properties[propertyIdx] = { ...properties[propertyIdx], ...updated, price: parseInt(updated.price), bedrooms: parseInt(updated.bedrooms) || 0, bathrooms: parseInt(updated.bathrooms) || 0, sqft: parseInt(updated.sqft) || 0, rating: parseFloat(updated.rating) || 0, reviews: parseInt(updated.reviews) || 0, amenities: updated.amenities ? updated.amenities.split(',').map(a => a.trim()) : [] }; saveProperties(); closeEditModal(); showMessage('propertiesMessage', '✅ Property updated successfully!', 'success'); applyFilters(); } } function openDeleteModal(propertyId) { propertyToDelete = propertyId; document.getElementById('deleteModal').classList.add('show'); } function closeDeleteModal() { document.getElementById('deleteModal').classList.remove('show'); propertyToDelete = null; } function confirmDelete() { if (propertyToDelete !== null) { properties = properties.filter(p => p.id !== propertyToDelete); saveProperties(); closeDeleteModal(); showMessage('propertiesMessage', '✅ Property deleted successfully!', 'success'); applyFilters(); } } // Close modals on escape key document.addEventListener('keydown', function(e) { if (e.key === 'Escape') { closeEditModal(); closeDeleteModal(); } });
